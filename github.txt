setting default branch ke main 
    cek list branch repositori
        git branch
        klo g ada isinya buat default branch ke main:
            git checkout -b main
    kalo udh terlanjur 
        cara1 master rename jd main dulu 
            git branch -m master main
            push perubahan: git push origin main
            apus branch master: git push origin --delete master
            update di local:
                git fetch origin
                git branch -u origin/main main
        cara2 buat main branch trs apus yg master 
            buat main branch: git checkout -b main
            push ke gh: git push origin main
